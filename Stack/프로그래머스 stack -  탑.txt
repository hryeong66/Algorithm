#include <string>
#include <vector>

using namespace std;

class Stack{
    int top;
    vector<int> s;
    
public: 
    Stack() : top(-1){ }
    bool isEmpty(){return top == -1;}
    
    void push(int k){
        s.push_back(k);
        top++;
    }
    
    int pop(){
        return s[top--];
    }
    
    int getTop(){return top;}
    
    void display(){
        printf("\n---------------------------\n");
        for(int i = 0; i < s.size(); i++)
            printf("%2d", s[i]);
        printf("\n---------------------------\n");
    }
};


vector<int> solution(vector<int> heights) {
    vector<int> answer;
    Stack input,output;
    
    for(int i = 0; i < heights.size(); i++)
        input.push(heights[i]);
    
    while(!input.isEmpty()){
        bool check = false;
        int val = input.pop();

        for(int i = input.getTop(); i >= 0; i--)
            if(heights[i] > val){
                output.push(i+1);
                printf("output %d 廓縑 菟橫馬 = %d " , i+1 ,heights[i] );
                check = true;
                break;
            }
    
        if(check == false)
            output.push(0);
     
        printf("\n output 頂辨");
        output.display();
    }
    
    for(int i = 0; i < heights.size(); i++)
        answer.push_back(output.pop());
    
    return answer;
}
